server {
    listen 80;
    server_name server.danieldgp.com www.danieldgp.com;

    # Redirect HTTP to HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name server.danieldgp.com;

    # SSL certs will be managed by certbot (see below)
    ssl_certificate /etc/letsencrypt/live/server.danieldgp.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/server.danieldgp.com/privkey.pem;

    # Proxy to the local Node server
    location / {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket support
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_buffering off;

        # Let backend decide CORS origin, but pass Origin through so backend can echo it
        proxy_set_header Origin $http_origin;
    }

    # socket.io polling location (optional explicit tuning)
    location /socket.io/ {
        proxy_pass http://127.0.0.1:3001/socket.io/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header Origin $http_origin;
        proxy_buffering off;
    }

    # Optional: don't override backend CORS headers; but if you must add them here:
    # add_header 'Access-Control-Allow-Origin' "$http_origin" always;
    # add_header 'Access-Control-Allow-Credentials' 'true' always;
}
